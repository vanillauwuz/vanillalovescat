ประเพศตัวแปรเริ่มต้นและวิธีใช้

1. char text ใช้ตัวแปร char เพื่อเก็บข้อมูลชนิดตัวอักขระตัวเดียวแต่ถ้าใส่ text[11] จะเก็บได้ 10 ตัว
2. int num   ใช้ตัวแปร int เพื่อเก็บข้อมูลเป็นเลขจำนวนเต็ม
3. float num ใช้ตัแปร float เพื่อเก็บข้อมูลชนิดเลขทศนิยม 6 ตําแหน่ง
3. double num ใช้ตัแปร double เพื่อเก็บข้อมูลชนิดเลขทศนิยม 12 ตําแหน่ง

#include <stdio.h>
#include <stdlib.h>

int main() {
    char Class = 'C', name[20] = "Program";
    int amount_book = 1001;
    float price_book = 170.50;
    printf("Welcome to BookStore\n");
    printf("%s %c Have %d book\n", name, Class, amount_book);
    printf("%s %c Is %.2f baht per book\n", name, Class, price_book);
    system("pause");
    return 0;
}
