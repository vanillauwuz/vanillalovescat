if_else 
โปรแกรมนี้เป็นโปรแกรมที่ใช้ให้ผู้ใช้ป้อนเพศของตนเองแล้วแสดงผลลัพธ์ว่าเป็นเพศชายหรือเพศหญิง

ส่วนประกอบของโปรแกรม:

ไลบรารี (Library) ที่ใช้งาน: stdio.h

ตัวแปรที่ใช้งาน:

1. gender: ตัวแปรชนิด char เพื่อเก็บค่าเพศที่ผู้ใช้ป้อน

โค้ดที่ใช้งาน:

1. printf(): ฟังก์ชันสำหรับแสดงผลข้อความหรือค่าตัวแปรในหน้าจอคอนโซล
2. scanf(): ฟังก์ชันสำหรับรับค่าที่ผู้ใช้ป้อนเข้ามา
3. if-else: โครงสร้างควบคุมที่ใช้ในการตรวจสอบเงื่อนไขและทำงานตามเงื่อนไข
#include <stdio.h>

int main() {
    char gender;
    printf("Enter your gender (M/F): ");
    scanf(" %c", &gender);
    if (gender == 'M' or gender == 'm' or gender == 'Male') {
        printf("Male\n");
    } else if (gender == 'F' or gender == 'f' or gender == 'Female') {
        printf("Female\n");
    } else {
        printf("Invalid input\n");
    }
    return 0;
}
